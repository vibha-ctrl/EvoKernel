```python
import torch
import triton
import triton.language as tl


@triton.jit
def _rope_kernel_2heads_unrolled(
    Q,
    K,
    Cos,
    Sin,
    Q_out,
    K_out,
    seq_len,
    n_heads,
    head_dim,
    stride_q_seq,
    stride_q_head,
    stride_k_seq,
    stride_k_head,
    stride_cos_seq,
    stride_sin_seq,
    BLOCK_SIZE: tl.constexpr,
    HALF_DIM: tl.constexpr,
):
    pid = tl.program_id(0)
    # Each block processes 2 heads
    seq_idx = pid // ((n_heads + 1) // 2)
    head_pair_idx = pid % ((n_heads + 1) // 2)
    head_idx0 = head_pair_idx * 2
    head_idx1 = head_idx0 + 1

    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < HALF_DIM

    # Compute base pointers for cos/sin (shared across heads in same position)
    base_cos = Cos + seq_idx * stride_cos_seq
    base_sin = Sin + seq_idx * stride_sin_seq

    # Load cos and sin values (shared for both heads)
    cos = tl.load(base_cos + cols, mask=mask, other=1.0).to(tl.float32)
    sin = tl.load(base_sin + cols, mask=mask, other=0.0).to(tl.float32)

    # Process first head
    base_q0 = Q + seq_idx * stride_q_seq + head_idx0 * stride_q_head
    base_k0 = K + seq_idx * stride_k_seq + head_idx0 * stride_k_head

    # Load Q components for head 0
    q0_0 = tl.load(base_q0 + cols, mask=mask, other=0.0).to(tl.float32)
    q0_1 = tl.load(base_q0 + cols + HALF_DIM, mask=mask, other=0.0).to(tl.float32)

    # Load K components for head 0
    k0_0 = tl.load(base_k0 + cols, mask=mask, other=0.0).to(tl.float32)
    k0_1 = tl.load(base_k0 + cols + HALF_DIM, mask=mask, other=0.0).to(tl.float32)

    # Unrolled RoPE for head 0
    rq0_0 = q0_0 * cos - q0_1 * sin
    rq0_1 = q0_0 * sin + q0_1 * cos
    rk0_0 = k0_0 * cos - k0_1 * sin
    rk0_1 = k0_0 * sin + k0_1 * cos

    # Store results for head 0
    base_qo0 = Q_out + seq_idx * stride_q_seq + head_idx0 * stride_q_head
    base_ko0 = K_out + seq_idx * stride_k_seq + head_idx0 * stride_k_head
    tl.store(base_qo0 + cols, rq0_0.to(tl.float16), mask=mask)
    tl.store(base_qo0 + cols + HALF_DIM, rq0_1.to(tl.float16), mask=mask)
    tl.store(base_ko0 + cols, rk0_0.to(tl.float16), mask=mask)
    tl.store(base_ko0 + cols + HALF_DIM, rk0_1.to(tl.float16), mask=mask)

    # Process second head if it exists
    head1_valid = head_idx1 < n_heads
    if head1_valid:
        base_q1 = Q + seq_idx * stride_q_seq + head_idx1 * stride_q_head
        base_k1 = K + seq_idx * stride_k_seq + head_idx1 * stride_k_head

        # Load Q components for head 1
        q1_0 = tl.load(base_q1 + cols, mask=mask, other=0.0).to(tl.float32)
        q1_1 = tl.load(base_q1 + cols + HALF_DIM, mask=mask, other=0.0).to(tl.float32)

        # Load K components for head 1
        k1_0 = tl.load(base_k1 + cols, mask=mask, other=0.0).to(tl.float32)
        k1_1 = tl.load(base_k1 + cols + HALF_DIM, mask=mask, other=0.0).to(tl.float32)

        # Unrolled RoPE for head 1
        rq1_0 = q1_0 * cos - q1_1 * sin
        rq1_1 = q1_0 * sin + q1_1 * cos
        rk1_0 = k1_0 * cos - k1_1 * sin
        rk1_1 = k1_0 * sin + k1_1 * cos

        # Store results for head 1
        base_qo1 = Q_out + seq_idx * stride_q_seq + head_idx1 * stride_q_head
        base_ko1 = K_out + seq_idx * stride_k_seq + head_idx1 * stride_k_head
        tl.store(base_qo1 + cols, rq1_0.to(tl.float16), mask=mask)
        tl.store(base_qo1 + cols + HALF_DIM, rq1_1.to(tl.float16), mask=mask)
        tl.store(base_ko1 + cols, rk1_0.to(tl.float16), mask=mask)
        tl.store(base_ko1 + cols + HALF_DIM, rk1_1.to(tl.float16), mask=mask)


def run(
    q: torch.Tensor,
    k: torch.Tensor,
    cos: torch.Tensor,
    sin: torch.Tensor,
) -> tuple[torch.Tensor, torch.Tensor]:
    assert q.is_cuda
    seq_len, n_heads, head_dim = q.shape
    half_dim = head_dim // 2

    q_out = torch.empty_like(q)
    k_out = torch.empty_like(k)

    # Compute strides
    stride_q_seq = q.stride(0)
    stride_q_head = q.stride(1)
    stride_k_seq = k.stride(0)
    stride_k_head = k.stride(1)
    stride_cos_seq = cos.stride(0)
    stride_sin_seq = sin.stride(0)

    # Each block processes 2 heads
    n_head_pairs = (n_heads + 1) // 2
    grid = (seq_len * n_head_pairs,)
    BLOCK_SIZE = 64

    _rope_kernel_2heads_unrolled[grid](
        q, k, cos, sin, q_out, k_out,
        seq_len, n_heads, head_dim,
        stride_q_seq, stride_q_head,
        stride_k_seq, stride_k_head,
        stride_cos_seq, stride_sin_seq,
        BLOCK_SIZE=BLOCK_SIZE,
        HALF_DIM=half_dim,
        num_warps=1,
        num_stages=1,
    )
    return q_out, k_out


KERNEL_TYPE = "rope"
DESCRIPTION = "Unrolled RoPE with 2 heads per block, BLOCK_SIZE=64, num_warps=1"
```
