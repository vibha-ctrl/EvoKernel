## Best Kernel Source Code

```python
import torch
import triton
import triton.language as tl


@triton.jit
def _rmsnorm_kernel_block_ptr_prefetch(
    X,
    W,
    Y,
    N,
    eps,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    
    # Create block pointers for more efficient memory access
    X_block_ptr = tl.make_block_ptr(
        base=X,
        shape=(N,),
        strides=(1,),
        offsets=(row * N,),
        block_shape=(BLOCK_SIZE,),
        order=(0,),
    )
    
    W_block_ptr = tl.make_block_ptr(
        base=W,
        shape=(N,),
        strides=(1,),
        offsets=(0,),
        block_shape=(BLOCK_SIZE,),
        order=(0,),
    )
    
    Y_block_ptr = tl.make_block_ptr(
        base=Y,
        shape=(N,),
        strides=(1,),
        offsets=(row * N,),
        block_shape=(BLOCK_SIZE,),
        order=(0,),
    )
    
    # Load data using block pointers with boundary check
    # With num_stages=4, the compiler will prefetch data ahead
    x = tl.load(X_block_ptr, boundary_check=(0,), padding_option="zero").to(tl.float32)
    w = tl.load(W_block_ptr, boundary_check=(0,), padding_option="zero").to(tl.float32)
    
    # Compute x^2 for variance calculation
    x_sq = x * x
    
    # Need to mask out padded values for correct sum
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < N
    x_sq = tl.where(mask, x_sq, 0.0)
    
    # Use tl.sum for reduction
    sum_sq = tl.sum(x_sq, axis=0)
    
    var = sum_sq / N
    rstd = tl.rsqrt(var + eps)
    
    y = (x * rstd * w).to(tl.float16)
    
    # Store result using block pointer
    tl.store(Y_block_ptr, y, boundary_check=(0,))


def run(x: torch.Tensor, weight: torch.Tensor, eps: float = 1e-5) -> torch.Tensor:
    assert x.is_cuda and weight.is_cuda
    M, N = x.shape
    y = torch.empty_like(x)
    BLOCK_SIZE = triton.next_power_of_2(N)
    _rmsnorm_kernel_block_ptr_prefetch[(M,)](
        x, weight, y, N, eps,
        BLOCK_SIZE=BLOCK_SIZE,
        num_warps=4,
        num_stages=4,  # Increased from 2 to 4 for better prefetching
    )
    return y


KERNEL_TYPE = "rmsnorm"
DESCRIPTION = "RMSNorm using tl.make_block_ptr with num_stages=4 for prefetching hints"
```
